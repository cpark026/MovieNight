228def get_results():
229 query = request.args.get("name", "").strip()
230 if not query:
231 return jsonify([])
232
233 encoded_query = urllib.parse.quote_plus(query)
234 search_url = f"https://api.themoviedb.org/3/search/movie?query={encoded_query}&page=1&include_adult=false"
235
236 headers = {
237 "accept": "application/json",
238 "Authorization": f"Bearer {TMDB_BEARER_TOKEN}"
239 }
240
241 try:
242 search_resp = requests.get(search_url, headers=headers)
243 search_resp.raise_for_status()
244 results = search_resp.json().get("results", [])
245 except Exception as e:
246 print("TMDb search API error:", e)
247 return jsonify([])
248
249 final_data = []
250
251 for item in results:
252 movie_id = item.get("id")
253 if not movie_id:
254 continue
255
256 detail_url = f"https://api.themoviedb.org/3/movie/{movie_id}?append_to_response=credits"
257 try:
258 detail_resp = requests.get(detail_url, headers=headers)
259 detail_resp.raise_for_status()
260 details = detail_resp.json()
261 except Exception as e:
262 print(f"TMDb detail API error for movie {movie_id}:", e)
263 continue
264
265 genres = "|".join([g.get("name") for g in details.get("genres", []) if g.get("name")])
266 prod_companies = "|".join([c.get("name") for c in details.get("production_companies", []) if c.get("name")])
267
268 credits = details.get("credits", {})
269 cast_list = []
270 for cast_member in credits.get("cast", []):
271 name_parts = cast_member.get("name", "").split(" ", 1)
272 first = name_parts[0] if len(name_parts) > 0 else ""
273 last = name_parts[1] if len(name_parts) > 1 else ""
274 cast_list.append(f"{first}|{last}")
275 cast_and_crew = ",".join(cast_list)
276
277 poster_path = details.get('poster_path')
278 final_data.append({
279 "id": details.get("id"),
280 "title": details.get("title"),
281 "adult": details.get("adult", False),
282 "genres": genres,
283 "overview": details.get("overview", ""),
284 "production_companies": prod_companies,
285 "cast_and_crew": cast_and_crew,
286 "rating_count": details.get("vote_count", 0),
287 "userRating": None,
288 "poster": f"https://image.tmdb.org/t/p/w500{poster_path}" if poster_path else "",
289 "poster_path": poster_path
290 })
291

33def extract_cast_names(cc_str):
34 """
35 Extract cast members from pipe-separated cast string.
36
37 Parses cast_and_crew format: "FirstName|LastName,FirstName|LastName,..."
38 Returns a set of lowercase "firstname lastname" pairs for easy comparison.
39
40 Args:
41 cc_str (str): Cast string in format "First|Last,First|Last,..."
42
43 Returns:
44 set: Set of lowercased "firstname lastname" pairs
45
46 Example:
47 >>> extract_cast_names("Tom|Hanks,Tim|Allen")
48 {'tom hanks', 'tim allen'}
49 """
50 if not cc_str:
51 return set()
52 names = set()
53 # Iterate through comma-separated cast entries
54 for entry in cc_str.split(","):
55 # Split by pipe to get first and last names
56 parts = entry.strip().split("|")
57 if len(parts) >= 2:
58 first, last = parts[0].strip().lower(), parts[1].strip().lower()
59 if first and last:
60 names.add(f"{first} {last}")
61 return names

64def extract_base_title(title):
65 """
66 Extract base franchise name from movie title.
67
68 Removes franchise indicators like "Part 2", "Returns", "Sequel", etc.
69 to identify movies from the same franchise.
70
71 Args:
72 title (str): Movie title string
73
74 Returns:
75 str: Lowercased base title without franchise indicators
76
77 Example:
78 >>> extract_base_title("Toy Story 2: The Sequel")
79 'toy story'
80 """
81 if not title:
82 return ""
83 # Remove "Part X" or "Vol X" patterns
84 title = re.sub(r'\s+(Part|Vol)\s+[IVX0-9]+', '', title, flags=re.IGNORECASE)
85 # Remove trailing numbers
86 title = re.sub(r'\s+\d+$', '', title)
87 # Remove sequel/return indicators
88 title = re.sub(r':\s+(The\s+)?Sequel|Returns|Reborn|Unleashed', '', title, flags=re.IGNORECASE)
89 # Remove parenthetical suffixes
90 title = re.sub(r'\s+\‍(.*?\‍)', '', title)
91 return title.strip().lower()

94def genre_similarity(user_genres, movie_genres):
95 """
96 Calculate Jaccard similarity between user's genres and movie's genres.
97
98 Jaccard similarity = |intersection| / |union|
99 Ranges from 0.0 (no overlap) to 1.0 (identical genres)
100
101 Args:
102 user_genres (str): Pipe-separated genre string from user profile
103 movie_genres (str): Pipe-separated genre string from candidate movie
104
105 Returns:
106 float: Jaccard similarity score [0.0, 1.0]
107 """
108 if not user_genres or not movie_genres:
109 return 0.0
110 # Convert pipe-separated strings to sets for intersection/union calculation
111 user_set = set(user_genres.split("|"))
112 movie_set = set(movie_genres.split("|"))
113 # Jaccard similarity = overlap / total unique
114 return float(len(user_set & movie_set)) / float(len(user_set | movie_set))

117def franchise_similarity(user_title, movie_title):
118 """
119 Check if two movies belong to the same franchise.
120
121 Uses extract_base_title to compare base franchise names.
122
123 Args:
124 user_title (str): Title of user's rated movie
125 movie_title (str): Title of candidate movie
126
127 Returns:
128 float: 1.0 if same franchise, 0.0 otherwise
129 """
130 # Extract base titles to compare franchises
131 user_base = extract_base_title(user_title)
132 movie_base = extract_base_title(movie_title)
133 # Return 1.0 if same franchise and not empty
134 if user_base == movie_base and user_base:
135 return 1.0
136 return 0.0

139def cast_similarity(user_names_str, movie_names_str):
140 """
141 Calculate cast overlap as a ratio of common actors.
142
143 Compares pre-extracted cast names. Returns ratio of overlapping actors
144 to total actors in the candidate movie.
145
146 Args:
147 user_names_str (str): Pipe-separated cast names from user's movies
148 movie_names_str (str): Pipe-separated cast names from candidate movie
149
150 Returns:
151 float: Ratio [0.0, 1.0] of movie actors present in user's cast
152 """
153 if not user_names_str or not movie_names_str:
154 return 0.0
155 # Convert pipe-separated strings to sets for overlap calculation
156 user_set = set(user_names_str.split("|"))
157 movie_set = set(movie_names_str.split("|"))
158 if not user_set or not movie_set:
159 return 0.0
160 # Return overlap ratio relative to movie's total cast
161 intersection = user_set & movie_set
162 return float(len(intersection)) / float(len(movie_set))

165def read_csv():
166 """
167 Load and preprocess CSV data into Spark DataFrame.
168
169 Performs the following operations:
170 1. Reads CSV with proper escape/quote handling for complex fields
171 2. Extracts cast names from cast_and_crew field
172 3. Pre-computes base titles for franchise matching
173 4. Splits genres and cast into arrays for efficient Spark operations
174 5. Cleans and casts numeric columns
175 6. Caches result for reuse
176
177 Returns:
178 DataFrame: Cached Spark DataFrame with preprocessed movie data
179
180 Global:
181 df_all: Set to the loaded and cached DataFrame
182 """
183 global df_all, spark
184
185 # Read CSV with special handling for multiline and quoted fields
186 df = spark.read.option("header", True) \
187 .option("multiLine", True) \
188 .option("escape", "\"") \
189 .option("quote", "\"") \
190 .csv(CSV_PATH)
191
192 # Extract cast names and convert to pipe-separated format
193 @udf(StringType())
194 def extract_names_udf(cc_str):
195 """Extract and join cast names from cast_and_crew string."""
196 names = extract_cast_names(cc_str)
197 return "|".join(sorted(names)) if names else ""
198
199 df = df.withColumn("cast_names", extract_names_udf(col("cast_and_crew")))
200
201 # Extract base franchise titles
202 @udf(StringType())
203 def base_title_udf(title):
204 """Extract base franchise title."""
205 return extract_base_title(title)
206
207 df = df.withColumn("base_title", base_title_udf(col("title")))
208
209 # Pre-split genres and cast names into arrays for efficient operations
210 df = df.withColumn("genres_array", split(col("genres"), "\\|")) \
211 .withColumn("cast_names_array", split(col("cast_names"), "\\|"))
212
213 # Clean and cast numeric columns
214 numeric_columns = ["avg_rating", "rating_count"]
215 for c in numeric_columns:
216 if c in df.columns:
217 df = df.withColumn(c, regexp_replace(col(c), r'[^0-9.\-]', "")) \
218 .withColumn(c, col(c).cast(FloatType()))
219 # Fill missing numeric values with 0.0
220 df = df.fillna({c: 0.0 for c in numeric_columns})
221
222 # Cache for performance
223 df_all = df.cache()
224 return df_all

255def get_top_recommendations(user_id, top_n=10):
256 """
257 Generate personalized recommendations for a user.
258
259 Algorithm:
260 1. Loads user's rated movies from SQLite
261 2. Builds weighted user profile combining genres, cast, and ratings
262 3. Filters out user's already-rated movies
263 4. Calculates similarity scores for remaining movies:
264 - Genre similarity (45% weight)
265 - Cast similarity (15% weight)
266 - Franchise similarity (5% weight)
267 - User's average rating norm (35% weight)
268 5. Returns top N movies by hybrid score
269
270 Args:
271 user_id (int): Database user ID
272 top_n (int): Number of recommendations to return (default 10)
273
274 Returns:
275 list: List of dicts with keys:
276 - title: Movie title
277 - cast_names: Pipe-separated cast string
278 - genre_sim, cast_sim, franchise_sim: Individual similarity scores
279 - user_rating_norm: User's average rating
280 - hybrid_score: Combined score [0.0, 1.0]
281 - cast_overlap: List of cast members in both user profile and movie
282 """
283 _timer_start = time.time()
284 global spark, df_all
285 # Initialize Spark if not already done
286 if df_all is None:
287 create_spark_and_load_data()
288
289 # Fetch user's rated movies from database
290 conn = sqlite3.connect(DB_PATH)
291 cur = conn.cursor()
292 cur.execute("SELECT * FROM movies WHERE user_id = ? AND userRating IS NOT NULL AND userRating > 0", (user_id,))
293 rows = cur.fetchall()
294 columns = [desc[0] for desc in cur.description]
295 conn.close()
296
297 # Convert rows to list of dicts
298 user_movies_list = [dict(zip(columns, row)) for row in rows]
299 if not user_movies_list:
300 print(f"No rated movies found for user {user_id}")
301 return []
302
303 # Extract cast names for each user movie
304 for movie_dict in user_movies_list:
305 names = extract_cast_names(movie_dict["cast_and_crew"])
306 movie_dict["cast_names"] = "|".join(sorted(names))
307
308 # Create Spark DataFrame from user movies
309 df_user = spark.createDataFrame(user_movies_list)
310 # Normalize ratings to [0, 1] and split genres/cast into arrays
311 df_user = df_user.withColumn("weight", col("userRating") / 10.0) \
312 .withColumn("cast_names_array", split(col("cast_names"), "\\|")) \
313 .withColumn("genres_array", split(col("genres"), "\\|"))
314
315 # Aggregate user profile: collect all genres and cast members
316 profile_agg = df_user.selectExpr(
317 "collect_list(id) as user_ids",
318 "collect_list(title) as user_titles",
319 "avg(weight * weight) / avg(weight) as weighted_avg_rating",
320 "concat_ws('|', collect_set(concat_ws('|', genres))) as all_genres",
321 "concat_ws('|', collect_set(concat_ws('|', cast_names))) as all_cast"
322 ).collect()[0]
323
324 # Extract profile components
325 user_ids_list = [str(uid) for uid in profile_agg.user_ids]
326 user_titles_list = profile_agg.user_titles
327 weighted_avg_rating = float(profile_agg.weighted_avg_rating) if profile_agg.weighted_avg_rating else 0.0
328
329 # Broadcast user profile to all workers for UDF computation
330 user_cast_str = profile_agg.all_cast if profile_agg.all_cast else ""
331 user_cast_array = [c.strip() for c in user_cast_str.split("|") if c.strip()]
332
333 bc_genres = spark.sparkContext.broadcast(profile_agg.all_genres)
334 bc_cast = spark.sparkContext.broadcast(profile_agg.all_cast)
335 bc_titles = spark.sparkContext.broadcast("|".join(user_titles_list))
336 bc_avg_rating = spark.sparkContext.broadcast(weighted_avg_rating)
337 bc_user_cast_array = spark.sparkContext.broadcast(user_cast_array)
338
339 # Filter out user's already-rated movies
340 df_filtered = df_all.filter(~col("id").isin(user_ids_list))
341
342 # UDF: Calculate genre similarity with user profile
343 @udf(FloatType())
344 def genre_sim_udf(genres_array):
345 """Calculate genre similarity with user profile."""
346 if not genres_array or not bc_genres.value:
347 return 0.0
348 user_set = set(bc_genres.value.split("|"))
349 movie_set = set([g.strip() for g in genres_array if g.strip()])
350 if not user_set or not movie_set:
351 return 0.0
352 return float(len(user_set & movie_set)) / float(len(user_set | movie_set))
353
354 # UDF: Calculate cast overlap ratio with user profile
355 @udf(FloatType())
356 def cast_sim_udf(cast_array):
357 """Calculate cast overlap ratio with user profile."""
358 if not cast_array or not bc_cast.value:
359 return 0.0
360 user_set = set([c.strip() for c in bc_cast.value.split("|") if c.strip()])
361 movie_set = set([c.strip() for c in cast_array if c.strip()])
362 if not user_set or not movie_set:
363 return 0.0
364 intersection = user_set & movie_set
365 return float(len(intersection)) / float(len(movie_set))
366
367 # UDF: Check if movie is in same franchise as any user movie
368 @udf(FloatType())
369 def franchise_sim_udf(movie_title):
370 """Check if movie is in same franchise as any user movie."""
371 if not movie_title or not bc_titles.value:
372 return 0.0
373 return 1.0 if any(franchise_similarity(t, movie_title) > 0.0 for t in bc_titles.value.split("|")) else 0.0
374
375 # Calculate similarity scores for all candidate movies
376 df_profile_sim = df_filtered \
377 .withColumn("genre_sim", genre_sim_udf(col("genres_array"))) \
378 .withColumn("cast_sim", cast_sim_udf(col("cast_names_array"))) \
379 .withColumn("franchise_sim", franchise_sim_udf(col("title"))) \
380 .withColumn("user_rating_norm", lit(bc_avg_rating.value)) \
381 .withColumn("cast_overlap_array", array_intersect(col("cast_names_array"), when(size(col("cast_names_array")) > 0, col("cast_names_array")))) \
382 .withColumn("hybrid_score",
383 0.45 * col("genre_sim") +
384 0.15 * col("cast_sim") +
385 0.05 * col("franchise_sim") +
386 0.35 * col("user_rating_norm")) \
387 .orderBy(col("hybrid_score").desc())
388
389 # Collect top N results from Spark cluster
390 results = df_profile_sim.select(
391 "title", "cast_names", "genre_sim", "cast_sim", "franchise_sim", "user_rating_norm", "hybrid_score", "cast_overlap_array"
392 ).limit(top_n).collect()
393
394 # Format results
395 result_list = []
396 for row in results:
397 rec_dict = row.asDict()
398 rec_dict["cast_overlap"] = sorted([c for c in rec_dict.get("cast_overlap_array", []) if c])
399 result_list.append(rec_dict)
400
401 _elapsed = time.time() - _timer_start
402 print(f"[TIMER] get_top_recommendations: {_elapsed:.2f}s")
403 return result_list

524def get_recommendations_by_most_common_genre(user_id, top_n=10):
525 """
526 Generate recommendations based on user's most frequently watched genre.
527
528 Algorithm:
529 1. Identifies the most common genre in user's rated movies
530 2. Filters movies containing that genre
531 3. Scores movies using:
532 - Genre match (70% weight) - 1.0 if contains main genre, 0.0 otherwise
533 - Average rating (30% weight)
534 4. Returns top N movies by combined score
535
536 Args:
537 user_id (int): Database user ID
538 top_n (int): Number of recommendations to return (default 10)
539
540 Returns:
541 list: List of dicts with keys:
542 - title: Movie title
543 - genres: Pipe-separated genre string
544 - avg_rating: Movie's average rating
545 - genre_match: 1.0 if contains main genre, 0.0 otherwise
546 - score: Combined score [0.0, 1.0]
547 """
548 _timer_start = time.time()
549 global spark, df_all
550 # Initialize Spark if not already done
551 if df_all is None:
552 create_spark_and_load_data()
553
554 # Fetch user's genres from database
555 conn = sqlite3.connect(DB_PATH)
556 cur = conn.cursor()
557 cur.execute("SELECT genres FROM movies WHERE user_id = ? AND genres IS NOT NULL AND genres != ''", (user_id,))
558 genre_rows = cur.fetchall()
559 conn.close()
560 if not genre_rows:
561 return []
562
563 # Find most common genre across all user movies
564 genre_count = {}
565 for (genre_str,) in genre_rows:
566 for g in genre_str.split("|"):
567 genre_count[g] = genre_count.get(g, 0) + 1
568
569 most_common_genre = max(genre_count, key=genre_count.get)
570 # Broadcast genre to all workers
571 bc_main_genre = spark.sparkContext.broadcast(most_common_genre)
572
573 # UDF: Check if movie contains the main genre
574 @udf(FloatType())
575 def single_genre_match(movie_genres):
576 """Check if movie contains the main genre."""
577 if not movie_genres:
578 return 0.0
579 return 1.0 if bc_main_genre.value in movie_genres.split("|") else 0.0
580
581 # Score all movies by genre match and rating
582 df_recs = df_all.withColumn("genre_match", single_genre_match(col("genres"))) \
583 .withColumn("score", 0.70 * col("genre_match") + 0.30 * (col("avg_rating") / 10.0)) \
584 .orderBy(col("score").desc())
585
586 # Collect top N results
587 results = df_recs.select("title", "genres", "avg_rating", "genre_match", "score").limit(top_n).collect()
588
589 _elapsed = time.time() - _timer_start
590 print(f"[TIMER] get_recommendations_by_most_common_genre: {_elapsed:.2f}s")
591 return [r.asDict() for r in results]
